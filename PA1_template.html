filled.activity <- activity

n <- length(activity$steps[is.na(activity$steps)])
mean.steps <- mean(activity$steps, na.rm = T)
sd.steps <- sd(activity$steps, na.rm = T)
values <- round(runif(n, min = 0, max = round(mean.steps + sd.steps)))
